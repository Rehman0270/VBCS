CREATE OR REPLACE FORCE EDITIONABLE VIEW "ADMIN"."PROPERTY_MANAGEMENT_DASHBOARD" ("MONTH", "NO_OF_CONTRACTS") DEFAULT COLLATION "USING_NLS_COMP"  AS 
  WITH month_ranges AS (
  SELECT 'current month' AS month_label, TRUNC(SYSDATE, 'MM') AS start_date, LAST_DAY(SYSDATE) AS end_date FROM DUAL
  UNION ALL
  SELECT 'last 3 months', TRUNC(ADD_MONTHS(SYSDATE, -2), 'MM'), LAST_DAY(SYSDATE) FROM DUAL
  UNION ALL
  SELECT 'last 6 months', TRUNC(ADD_MONTHS(SYSDATE, -5), 'MM'), LAST_DAY(SYSDATE) FROM DUAL
  UNION ALL
  SELECT 'last 9 months', TRUNC(ADD_MONTHS(SYSDATE, -8), 'MM'), LAST_DAY(SYSDATE) FROM DUAL
),
contract_data AS (
  SELECT
    CREATION_DATE,
    CONTRACT_NUMBER
  FROM ADMIN.PROPERTY_MANAGEMENT_HEADERS
  WHERE CREATION_DATE >= TRUNC(ADD_MONTHS(SYSDATE, -8), 'MM')
)
SELECT
  mr.month_label AS MONTH,
  COUNT(DISTINCT cd.CONTRACT_NUMBER) AS NO_OF_CONTRACTS
FROM
  month_ranges mr
LEFT JOIN contract_data cd
  ON cd.CREATION_DATE BETWEEN mr.start_date AND mr.end_date
GROUP BY
  mr.month_label
ORDER BY
  CASE mr.month_label
    WHEN 'current month' THEN 1
    WHEN 'last 3 months' THEN 2
    WHEN 'last 6 months' THEN 3
    WHEN 'last 9 months' THEN 4
  END
;
